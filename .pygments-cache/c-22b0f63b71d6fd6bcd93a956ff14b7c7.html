int findPalindromicString(char *s){
    char *sp=s,*ep=s+1;
    char *tmpSp,*tmpEp;
    bool isPalindromic=true;
    int result =0;
    int tmpResult =0;
    while (*sp)
    {
        while(*ep){
            for(tmpSp=sp,tmpEp=ep;tmpSp<tmpEp;tmpSp++,tmpEp--){
                if(*tmpSp != *tmpEp){
                    isPalindromic = false;
                    break;
                }
            }
            if(isPalindromic){
                for(tmpSp = sp;tmpSp<=ep;tmpSp++){
                    tmpResult++;
                    printf("%c",*tmpSp);
                }
                printf("\n");
                if(tmpResult > result){
                    result = tmpResult;
                }
            }
            isPalindromic = true;
            ep++;
        }
        sp++;
        ep = sp + 1;
        isPalindromic = true;
        tmpResult = 0;
    }
    return result;    
}
