int lengthOfLongestSubstring(char* s) {
    char *sp;
    int array[128]={0};
    int i,j,len,max;
    sp=s;
    len=0;
    max=0;
    while(*sp){
        i=*sp;
        if(array[i]==0){
            len++;
            array[i]=len;
            sp++;
        }else{
            sp-=len-array[i];
            if(len>max){
                max=len;
            }
            len=0;
            for(j=0;j<128;j++){
                array[j]=0;
            }
        } 
    }
    if(len>max){
        max=len;
    }
    return max;
    
}
