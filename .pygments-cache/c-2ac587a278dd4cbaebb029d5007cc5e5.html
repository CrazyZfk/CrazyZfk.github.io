int maxArea(int* height, int heightSize){
    int *first,*end,maxArea=0;
    int realHeight,tmpArea=0;
    first = &height[0];
    end = &height[heightSize-1];
    while (first <= end){
        realHeight = (*first)>=(*end)?(*end):(*first);
        tmpArea = realHeight * (end - first);
        if(tmpArea>maxArea){
            maxArea = tmpArea;
        }
        if(*first>=*end){
            end--;
        } else{
            first++;
        }
    }
    return maxArea;
}
