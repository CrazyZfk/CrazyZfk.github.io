for(index=0;index<size;index++){
    if(str[index]==' '){
        if(beg==-1) {
            continue;
        }else{
            break;
        }
    }
    if(str[index]>='0'&&str[index]<='9'&&beg==-1){
        beg = index;
        end = index;
        continue;
    }
    if((str[index]=='+'||str[index]=='-')&&beg==-1){
        beg = index;
        end = index;
        continue;
    }
    if(str[index]<'0'||str[index]>'9'){
        if(beg==-1){
            return 0;
        } else{
            end = index-1;
            break;
        }
    }
    end = index;
}
if(beg<0||(beg==end)&&(str[beg]=='+'||str[beg]=='-')){
    return 0;
}
